Rather than coding this as a command-line switch to WeiDU, I decided to write a command-line program that does this. If and when I decide to pick up on this again, I'll think about adding the functionality of uninstalling specific mods (or components) only.
However, this is decent enough for a clear-my-install-and-backup-directories-while-I-get-a-coffee functionality.

Code:
#include <stdio.h>
#include <stdlib.h>
#include <strings.h>

#define ALLOC(x) (x *) malloc(sizeof(x))
#define ALLOCX(x,y) (x *) malloc((y)*sizeof(x))

#ifdef __DEBUG
  #define FREE(x) { printf("Freeing %s in %s at %d\n",#x,__FILE__,__LINE__);\
  free (x); \
   printf("Freed succesfully %s in %s at %d\n",#x,__FILE__,__LINE__); }
#endif

#ifndef __DEBUG
  #define FREE(x) free(x);
#endif

struct two_linked_list
{
  struct two_linked_list * before;
  char * string_start;
  struct two_linked_list * after;
};

typedef struct two_linked_list Two_linked_list;

int not_found(struct two_linked_list * begin, char * the_string)
{
  struct two_linked_list * position = begin;
  if (!strcmp(position->string_start,the_string))
  {
    return 0;
  }
  else
  {
    while (position->after != NULL)
    {
      position = position->after;
      if (!strcmp(position->string_start,the_string))
      {
        return 0;
      }
    }
  }
  return 1;
}

struct two_linked_list * list_memorize(struct two_linked_list * begin, struct two_linked_list * end, char * thestring)
/* returns the new ending */
{
  char * pc;
  for (pc = thestring  + 1; *pc != '~'; pc++);
  *pc = '\0';
  *(pc+1) = '\0';
//   printf("%s\n\n",thestring + 1);
  if (not_found (begin,thestring))
  {
    end->after = ALLOC(Two_linked_list);
    end->after->before = end;
    end->after->after = NULL;
    end->after->string_start = ALLOCX(char,strlen(thestring)+1);
    sprintf(end->after->string_start,"%s",thestring);
    return end->after;
  }
  else
  {
    return end;
  }
}

struct two_linked_list * parse_log(struct two_linked_list * begin)
/* returns the ending */
{
  FILE * log;
  char buffstring[1000];
  char firstchar;
  char * pc;
  int done = 0;
  struct two_linked_list * end = begin;
  log = fopen("weidu.log","r");
  if (log == NULL)
  {
    printf("Unable to open the log file\n");
    exit(1);
  }
  while(fgets(buffstring,999,log)!=NULL)
  {
    firstchar = buffstring[0];
    if (firstchar != '~') continue;
    if (!done)
    {
      done ++;
      for (pc = buffstring + 1; *pc != '~';pc++);
      *pc = '\0';
      begin->string_start = ALLOCX(char,strlen(buffstring)+3);
      sprintf(begin->string_start,"%s",buffstring);
      /* ((begin->string_start)+1) contains exactly the tp2 string */
    }
    else
    {
    &n bsp; end = list_memorize(begin,end,buffstring);
    }
  }
  return end;
}

void uninstall_mods_internal(struct two_linked_list * end)
{
  struct two_linked_list * pos;
  char torun[1000];
//   int i;
  sprintf(torun,"weidu %s --uninstall --tlkout dialog.tlk --ftlkout dialogF.tlk --autolog\n",end->string_start+1);
  printf("%s",torun);
#ifndef __DEBUG
  system(torun);
#endif
//   for (i = 0; i < strlen (torun);i++) printf("%2x %c\n",torun[i],torun[i]);
  FREE(end->string_start);
  pos=end->before;
  FREE(end);
  while(pos != NULL)
  {
    sprintf(torun,"weidu %s --uninstall --tlkout dialog.tlk --ftlkout dialogF.tlk --autolog\n",pos->string_start+1);
    printf("%s",torun);
  #ifndef __DEBUG
    system(torun);
  #endif
//     for (i = 0; i < strlen (torun);i++) printf("%2x %c\n",torun[i],torun[i]);
    FREE(pos->string_start);
    if (pos->before != NULL)
    {
      pos=pos->before;
      FREE(pos->after);
    }
    else
    {
      FREE(pos);
      pos=NULL;
    }
  }
}


int main()
{
  struct two_linked_list * beginning = ALLOC(Two_linked_list);
  struct two_linked_list * end = beginning;
  beginning->before = NULL;
  beginning->after = NULL;
  end=parse_log(beginning);
  uninstall_mods_internal(end);
  /* will also free() the memory */
  return 0;
}

If you have access to a C compiler, simply compile and run this program.
[pimping cygwin]
If you don't have a compiler, a really great and free CLI compiler is gcc (should be included with Linux, dunno about MacOS-X); under windows, I suggest you strongly to download http://www.cygwin.com cygwin and add to the base selection the gcc, gcc-core and gcc-g++ package (from the devel directory) to get access to gcc. (under a standard installation, the path to gcc is c:\cygwin\usr\bin\gcc.exe; if you use cygwin's bash rather than DOS prompt you'll have direct access to it via $PATH).
[/pimping cygwin]
